#
#   conda create -n penguins-thermal python=3.11
#   conda activate penguins-thermal

#
# METHOD 2: System GDAL + pip (Advanced)
# ---------------------------------------
# Install GDAL system libraries first:
#
#   macOS:   brew install gdal
#   Ubuntu:  sudo apt-get install gdal-bin libgdal-dev
#   Fedora:  sudo dnf install gdal gdal-devel
#
# Then install Python bindings (match GDAL version!):
#
#   gdal==$(gdal-config --version)
#   pip install gdal==$GDAL_VERSION
#   pip install rasterio pyproj geopandas
